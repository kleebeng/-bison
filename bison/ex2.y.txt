int yylex()
{
    char t;
 while(1){
  t = getchar();
  if(t==' ' || t =='\t'|| t == '\n')
      {}
  else if(t == '+'){
   return ADD;
  }
  else if(t == '*'){
   return MUL;
  }
  else if(t == '-'){
   return SUB;
  }
  else if(t == '/'){
   return DIV;
  }
  else if(t == '('){
   return LB;
  }
  else if(t == ')'){
   return RB;
  }
  else if(t>='0'&amp;&amp;t<='9'){
   int ti = 0;  
   while (t>='0'&amp;&amp;t<='9'){
                numStr[ti]=t;
                t = getchar();
                ti++;
            }
            numStr[ti] = '\0';
            yylval = numStr;
            ungetc(t,stdin);
            return NUMBER;
  }
        else if(('a'<=t&amp;&amp;t<='z')||('A'<=t&amp;&amp;'Z'>=t)||(t=='_')){
            int ti=0;
            while(('a'<=t&amp;&amp;t<='z')||('A'<=t&amp;&amp;'Z'>=t)||(t=='_')||(t>='0'&amp;&amp;t<='9')){
                idStr[ti] = t;
                ti++;
                t=getchar(); 
            }
            idStr[ti] = '\0';
            yylval = idStr;
            ungetc(t,stdin);
            return ID;   
        }
        else{
   return t;
  }
 }
}
